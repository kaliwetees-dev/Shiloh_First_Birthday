var sheetName = 'Sheet1'; var scriptProp = PropertiesService.getScriptProperties(); function intialSetup () { var activeSpreadsheet = SpreadsheetApp.getActiveSpreadsheet(); scriptProp.setProperty('key', activeSpreadsheet.getId()); } /** * Handles checking if a guest has already RSVP'd. * Returns their reserved counts if they exist. */ function doGet(e) { var userEmail = e.parameter.checkEmail; if (userEmail) { var doc = SpreadsheetApp.openById(scriptProp.getProperty('key')); var sheet = doc.getSheetByName(sheetName); var data = sheet.getDataRange().getValues(); var headers = data[0]; var gmailIdx = headers.indexOf('Gmail'); var adultsIdx = headers.indexOf('Adults'); var kidsIdx = headers.indexOf('Kids'); var foundRow = null; for (var i = 1; i < data.length; i++) { if (data[i][gmailIdx] == userEmail) { foundRow = data[i]; break; } } if (foundRow) { return ContentService.createTextOutput(JSON.stringify({ exists: true, adults: foundRow[adultsIdx] || 0, kids: foundRow[kidsIdx] || 0 })).setMimeType(ContentService.MimeType.JSON); } else { return ContentService.createTextOutput(JSON.stringify({exists: false})) .setMimeType(ContentService.MimeType.JSON); } } } /** * Saves initial RSVP data from the website. */ function doPost (e) { var lock = LockService.getScriptLock(); lock.tryLock(10000); try { var doc = SpreadsheetApp.openById(scriptProp.getProperty('key')); var sheet = doc.getSheetByName(sheetName); var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; var nextRow = sheet.getLastRow() + 1; var newRow = headers.map(function(header) { return header === 'Timestamp' ? new Date() : e.parameter[header] || ""; }); sheet.getRange(nextRow, 1, 1, newRow.length).setValues([newRow]); return ContentService.createTextOutput(JSON.stringify({ 'result': 'success' })).setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } } /** * AUTOMATION: Sends free email notifications. * Trigger this every 1 minute. */ function processConfirmations() { var doc = SpreadsheetApp.openById(scriptProp.getProperty('key')); var sheet = doc.getSheetByName(sheetName); var data = sheet.getDataRange().getValues(); var headers = data[0]; var gmailIdx = headers.indexOf('Gmail'); var adultsIdx = headers.indexOf('Adults'); var kidsIdx = headers.indexOf('Kids'); var confirmIdx = headers.indexOf('Confirmation'); // Loop through rows (skip header) for (var i = 1; i < data.length; i++) { var row = data[i]; var guestEmail = row[gmailIdx]; var adults = row[adultsIdx]; var kids = row[kidsIdx]; var confirmationStatus = row[confirmIdx]; // CONDITION: Adults AND Kids are filled, but Confirmation is still blank if (adults !== "" && kids !== "" && confirmationStatus.toString().trim() === "") { var subject = "RSVP Confirmed: Shiloh's 1st Birthday!"; var message = "Hi! \n\nThank you for confirming your attendance. We have reserved " + adults + " adult(s) and " + kids + " kid(s) for you. \n\n" + "We are so excited to see you beary soon at Shiloh's 1st Birthday! \n\n" + "Location: Malana's Residence, Sta. Cruz, Dapitan City \n" + "Time: 3:00 PM - 6:00 PM"; try { // Send the free email MailApp.sendEmail(guestEmail, subject, message); // Mark as "Yes" in the spreadsheet so it doesn't send again sheet.getRange(i + 1, confirmIdx + 1).setValue("Yes"); } catch (err) { Logger.log("Email failed for: " + guestEmail + " Error: " + err); } } } }
